在线预览ppt
wx.downloadFile({
  url: 'https://your-cdn.com/path/to/123.pptx', // HTTPS链接
  success: function (res) {
    const filePath = res.tempFilePath
    wx.openDocument({
      filePath: filePath,
      fileType: 'pptx',
      success: function (res) {
        console.log('打开文档成功')
      },
      fail: function (err) {
        console.error('打开文档失败:', err)
      }
    })
  },
  fail: function (err) {
    console.error('下载文件失败:', err)
  }
})
